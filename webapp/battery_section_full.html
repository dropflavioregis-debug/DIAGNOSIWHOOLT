// firmware/src/vehicle/battery_reader.cpp // Legge SOC, SOH, temp celle via UDS SID 0x22 #include "battery_reader.h" #include "../uds/uds_client.h" // DID specifici Renault Zoe ZE50 (da OBDb JSON) #define DID_SOC 0x2002 #define DID_SOH 0x2006 #define DID_CELL_TEMPS 0x4003 #define DID_PACK_VOLTAGE 0x2001 #define ECU_BMS 0x7EC BatteryData BatteryReader::read() { BatteryData data = {}; // 1. Apri sessione diagnostica extended uds.sessionControl(0x03); delay(50); // 2. Leggi SOC uint8_t buf[8]; if (uds.readDID(ECU_BMS, DID_SOC, buf)) { data.soc = buf[1] * 0.5f; // formula da libreria JSON } // 3. Leggi SOH if (uds.readDID(ECU_BMS, DID_SOH, buf)) { data.soh = buf[1] * 0.5f; } // 4. Leggi temperatura 96 celle (multi-frame ISO-TP) uint8_t tempBuf[128]; if (uds.readDIDLong(ECU_BMS, DID_CELL_TEMPS, tempBuf)) { for (int i = 0; i < 96; i++) { data.cell_temps[i] = tempBuf[i] - 40; // offset standard } } // 5. Invia a Vercel via WiFi api.postBattery(data); return data; }
// web/app/api/battery/route.ts import { createClient } from '@supabase/supabase-js' import { NextRequest, NextResponse } from 'next/server' export async function POST(req: NextRequest) { const supabase = createClient( process.env.SUPABASE_URL!, process.env.SUPABASE_SERVICE_KEY! ) const body = await req.json() // body: { device_id, session_id, soc, soh, pack_voltage, // cell_temps: number[96], recorded_at } // Salva lettura principale const { data, error } = await supabase .from('battery_readings') .insert({ session_id: body.session_id, soc: body.soc, soh: body.soh, pack_voltage: body.pack_voltage, cell_temps: body.cell_temps, // JSONB array recorded_at: body.recorded_at ?? new Date().toISOString() }) if (error) return NextResponse.json({ error }, { status: 500 }) return NextResponse.json({ ok: true, id: data?.[0]?.id }) } export async function GET(req: NextRequest) { const sessionId = req.nextUrl.searchParams.get('session_id') const { data } = await supabase .from('battery_readings') .select('*') .eq('session_id', sessionId) .order('recorded_at', { ascending: false }) .limit(100) return NextResponse.json(data) }
-- supabase/migrations/006_battery_readings.sql CREATE TABLE battery_readings ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), session_id UUID REFERENCES sessions(id) ON DELETE CASCADE, -- Parametri principali soc FLOAT NOT NULL, -- % 0-100 soh FLOAT, -- % 0-100 pack_voltage FLOAT, -- Volt pack_current FLOAT, -- Ampere -- Celle (JSONB array di 96 float) cell_temps JSONB, -- [22.1, 23.5, ...] cell_voltages JSONB, -- [3.71, 3.72, ...] -- Aggregati pre-calcolati (per query veloci) temp_min FLOAT, temp_max FLOAT, temp_avg FLOAT, volt_min FLOAT, volt_max FLOAT, volt_delta FLOAT, -- max - min celle recorded_at TIMESTAMPTZ DEFAULT NOW() ); -- Indice time-series per grafici storici CREATE INDEX idx_batt_session_time ON battery_readings(session_id, recorded_at DESC); -- Abilita Realtime per dashboard live ALTER TABLE battery_readings REPLICA IDENTITY FULL; SELECT supabase_realtime.enable_realtime('battery_readings');
1
ESP32 legge batteria ogni secondo TWAI + UDS
SID 0x22 → DID 0x2002 (SOC) + 0x2006 (SOH) + 0x4003 (96 temp). ISO-TP gestisce multi-frame. Dati raw → struttura BatteryData.
2
POST a Vercel WiFi · HTTPS
JSON compresso (gzip) con API key header. Vercel autentica, valida schema Zod, calcola aggregati (min/max/delta).
3
Insert in Supabase PostgreSQL
Row inserita in battery_readings. Supabase Realtime notifica tutti i client connessi via WebSocket entro ~50ms.
4
Dashboard aggiornata live Next.js · useEffect
Il hook useRealtimeBattery() riceve il nuovo row → aggiorna stato React → mappa celle e grafici ri-renderizzati senza refresh.
5
Anomalia rilevata → Claude API Trigger automatico
Se temp_delta > 2°C o soh_delta < −0.5% → POST a /api/analyze con i dati → Claude risponde con diagnosi in italiano.
