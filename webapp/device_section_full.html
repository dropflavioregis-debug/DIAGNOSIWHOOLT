// firmware/src/ota/ota_updater.cpp // OTA: ESP32 scarica nuovo firmware da Vercel // Controllato ogni 24h o su richiesta dalla dashboard #include <HTTPUpdate.h> void OtaUpdater::checkAndUpdate() { // 1. Chiedi al server se c'è una versione nuova HTTPClient http; http.begin(nvs.get("server_url") + "/api/firmware/latest"); http.addHeader("X-Device-ID", nvs.get("device_id")); http.addHeader("X-FW-Version", FIRMWARE_VERSION); http.GET(); String body = http.getString(); DynamicJsonDocument doc(256); deserializeJson(doc, body); if (!doc["update_available"].as<bool>()) { Serial.println("[OTA] Firmware aggiornato"); return; } String bin_url = doc["url"].as<String>(); Serial.printf("[OTA] Download %s\n", bin_url.c_str()); // 2. // firmware/src/can/can_driver.cpp // Inizializzazione TWAI nativo ESP32-S3 // SN65HVD230: TX→GPIO17, RX→GPIO18, VCC→3.3V, GND→GND // OBD2: CAN-H→CANH, CAN-L→CANL, pin16→DC-DC→3.3V #include "driver/twai.h" bool CanDriver::begin(int speed_kbps) { twai_timing_config_t timing; switch(speed_kbps) { case 500: timing = TWAI_TIMING_CONFIG_500KBITS(); break; case 250: timing = TWAI_TIMING_CONFIG_250KBITS(); break; case 1000: timing = TWAI_TIMING_CONFIG_1MBITS(); break; default: timing = TWAI_TIMING_CONFIG_500KBITS(); } twai_general_config_t g_config = TWAI_GENERAL_CONFIG_DEFAULT( (gpio_num_t)17, // TX pin (gpio_num_t)18, // RX pin TWAI_MODE_NORMAL ); twai_filter_config_t f_config = TWAI_FILTER_CONFIG_ACCEPT_ALL(); if (twai_driver_install(&g_config, &timing, &f_config) != ESP_OK) { Serial.println("[CAN] Install fallito"); return false; } if (twai_start() != ESP_OK) { Serial.println("[CAN] Start fallito"); return false; } Serial.printf("[CAN] OK — %d kbps — TX:17 RX:18\n", speed_kbps); return true; } bool CanDriver::send(uint32_t id, uint8_t* data, uint8_t len) { twai_message_t msg = {}; msg.identifier = id; msg.data_length_code = len; memcpy(msg.data, data, len); return twai_transmit(&msg, pdMS_TO_TICKS(100)) == ESP_OK; } bool CanDriver::receive(twai_message_t* msg, uint32_t timeout_ms) { return twai_receive(msg, pdMS_TO_TICKS(timeout_ms)) == ESP_OK; }
