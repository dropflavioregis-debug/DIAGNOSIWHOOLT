// firmware/src/uds/dtc_reader.cpp // Legge DTC da TUTTE le ECU — broadcast 0x7DF + indirizzamento diretto struct DtcEntry { char code[7]; // es. "P0AA6" uint8_t status; // byte di stato UDS uint8_t ecu_addr; // ECU sorgente }; // Indirizzi ECU da libreria JSON (caricata da SPIFFS) uint8_t ECU_ADDRS[] = { 0x7E0, // ECM/PCM 0x7E4, // BMS batteria 0x7A0, // BCM body 0x760, // ABS/ESP 0x744, // OBC caricatore 0x77E, // HVAC }; void DtcReader::scanAll(std::vector<DtcEntry>& out) { for (uint8_t addr : ECU_ADDRS) { // SID 0x19 sub 0x02 = reportDTCByStatusMask // 0xFF = tutti i DTC indipendentemente dallo stato uint8_t req[] = {0x03, 0x19, 0x02, 0xFF}; uint8_t resp[256]; uint16_t len = 0; if (!isotp.sendAndReceive(addr, addr+0x08, req, 4, resp, &len)) continue; // ECU non risponde // Parsing risposta: byte 1=SID+0x40, byte 2=sub, byte 3=status // poi gruppi di 4 byte: [B3][B2][B1][status] for (uint16_t i = 4; i+3 < len; i+=4) { DtcEntry e; parseDtcCode(resp[i], resp[i+1], resp[i+2], e.code); e.status = resp[i+3]; e.ecu_addr = addr; out.push_back(e); } } } void DtcReader::clearAll() { for (uint8_t addr : ECU_ADDRS) { // SID 0x14 = ClearDiagnosticInformation // 0xFFFFFF = cancella tutti i gruppi DTC uint8_t req[] = {0x04, 0x14, 0xFF, 0xFF, 0xFF}; isotp.send(addr, req, 5); delay(100); // attesa risposta positiva 0x54 } }
// web/app/api/dtc/route.ts — riceve DTC da ESP32, arricchisce con DB export async function POST(req: NextRequest) { const { session_id, device_id, dtcs } = await req.json() // dtcs: [{ code, status_byte, ecu_address, recorded_at }] const supabase = createClient(url, key) // 1. Arricchisci ogni DTC con descrizione dal DB const codes = dtcs.map((d: any) => d.code) const { data: descriptions } = await supabase .from('dtc') .select('code, description_it, severity, system, possible_causes') .in('code', codes) const enriched = dtcs.map((d: any) => ({ ...d, session_id, ...descriptions?.find(x => x.code === d.code), status: parseStatusByte(d.status_byte), // active = bit 0, pending = bit 5, confirmed = bit 6 })) // 2. Insert in Supabase await supabase.from('dtc_readings').insert(enriched) // 3. Se DTC critici → trigger analisi Claude const critical = enriched.filter(d => d.severity === 'critical') if (critical.length > 0) { await fetch('/api/analyze', { method: 'POST', body: JSON.stringify({ session_id, dtcs: critical }) }) } return NextResponse.json({ ok: true, enriched_count: enriched.length }) }

// web/app/api/analyze/route.ts — Claude analizza i DTC in italiano const SYSTEM = `Sei un esperto diagnostico di veicoli elettrici. Ricevi DTC letti via UDS da ESP32. Rispondi SOLO in JSON: { "summary": "frase breve stato generale", "dtcs": [{ "code": "P0AA6", "spiegazione": "cosa significa in italiano semplice", "urgenza": "immediata|entro_settimana|monitorare", "cause_probabili": ["causa 1", "causa 2"], "azione": "cosa fare concretamente" }], "raccomandazione_generale": "..." }` export async function POST(req: NextRequest) { const { session_id, dtcs, vehicle } = await req.json() const response = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'x-api-key': process.env.ANTHROPIC_API_KEY!, 'anthropic-version': '2023-06-01', 'content-type': 'application/json' }, body: JSON.stringify({ model: 'claude-sonnet-4-20250514', max_tokens: 1000, system: SYSTEM, messages: [{ role: 'user', content: `Veicolo: ${vehicle.make} ${vehicle.model} ${vehicle.year} DTC rilevati: ${JSON.stringify(dtcs, null, 2)}` }] }) }) const data = await response.json() const analysis = JSON.parse(data.content[0].text) // Salva analisi in Supabase await supabase.from('sessions') .update({ ai_diagnosis: analysis }) .eq('id', session_id) return NextResponse.json(analysis) }
1
ESP32 scansiona tutte le ECU UDS SID 0x19
Loop su 6–11 indirizzi ECU (da JSON libreria). Ogni ECU: SID 0x19 sub 0x02 mask 0xFF. Risposta analizzata in gruppi da 4 byte. Timeout 200ms per ECU.
2
Parsing codici ISO 15031-6
Byte 1 bit 7-6 → tipo (P/C/B/U). Bit 5-4 → categoria. Byte 2-3 → 4 cifre hex. Status byte → flags active/pending/confirmed.
3
POST a /api/dtc Vercel Edge
Array JSON con codici raw + ecu_address + status_byte. Server arricchisce con descrizioni da tabella dtc in Supabase.
4
Insert + Realtime Supabase
Righe inserite in dtc_readings. Supabase notifica dashboard via WebSocket. Se severity=critical → trigger automatico Claude API.
5
Analisi Claude → Supabase JSON strutturato
Claude risponde in JSON con spiegazione, urgenza e azioni per ogni DTC. Salvato in sessions.ai_diagnosis. Dashboard mostra live.
