// firmware/src/session_manager.cpp // Gestisce ciclo vita sessione sull'ESP32 // Sessione = dal momento in cui auto si accende a quando si spegne class SessionManager { String session_id; uint32_t started_ms; float soc_start, soc_end; uint32_t km_start; public: void onIgnitionOn() { // 1. Genera UUID locale (usato offline finché WiFi non disponibile) session_id = generateUUID(); started_ms = millis(); // 2. Leggi stato iniziale via UDS soc_start = uds.readSOC(); km_start = uds.readOdometer(); // 3. Apri sessione sul server (async, non blocca) api.postAsync("/api/sessions/open", { "session_id": session_id, "device_id": nvs.get("device_id"), "vehicle_id": nvs.get("vehicle_id"), "soc_start": soc_start, "km_start": km_start, "started_at": getISOTime() }); // 4. Avvia loop lettura ogni 1s xTaskCreate(readLoop, "ReadLoop", 8192, this, 5, &readTaskHandle); } void onIgnitionOff() { // Ferma il loop vTaskDelete(readTaskHandle); soc_end = uds.readSOC(); float km_end = uds.readOdometer(); // Chiudi sessione sul server con statistiche finali api.post("/api/sessions/close", { "session_id": session_id, "soc_end": soc_end, "km_driven": km_end - km_start, "duration_s": (millis() - started_ms) / 1000, "ended_at": getISOTime() }); session_id = ""; } static void readLoop(void* arg) { SessionManager* self = (static_cast<SessionManager*>(arg)); while(true) { self->readAndPost(); vTaskDelay(1000 / portTICK_PERIOD_MS); } } };
// web/app/api/sessions/open/route.ts export async function POST(req: NextRequest) { const body = await req.json() const { data, error } = await supabase .from('sessions') .insert({ id: body.session_id, // UUID generato dall'ESP32 device_id: body.device_id, vehicle_id: body.vehicle_id, soc_start: body.soc_start, km_start: body.km_start, started_at: body.started_at, status: 'active' }) .select().single() return NextResponse.json({ ok: !error, session: data }) } // web/app/api/sessions/close/route.ts export async function POST(req: NextRequest) { const { session_id, soc_end, km_driven, duration_s } = await req.json() // Calcola kWh consumati (SOC delta * capacità pack) const { data: sess } = await supabase .from('sessions').select('soc_start,vehicle_id') .eq('id', session_id).single() const { data: veh } = await supabase .from('vehicles').select('battery_kwh') .eq('id', sess.vehicle_id).single() const soc_delta = sess.soc_start - soc_end const kwh_used = (soc_delta / 100) * veh.battery_kwh const kwh_100km = km_driven > 0 ? (kwh_used / km_driven) * 100 : null await supabase.from('sessions').update({ soc_end, km_driven, duration_s, kwh_used, kwh_per_100km: kwh_100km, status: 'completed', ended_at: new Date().toISOString() }).eq('id', session_id) // Trigger analisi AI fine sessione await fetch('/api/ai/session-report', { method: 'POST', body: JSON.stringify({ session_id }) }) return NextResponse.json({ ok: true }) }
-- supabase/migrations/001_sessions.sql CREATE TABLE sessions ( id UUID PRIMARY KEY, -- generato dall'ESP32 device_id TEXT NOT NULL, vehicle_id UUID REFERENCES vehicles(id), -- Stato status TEXT DEFAULT 'active', -- active | completed | error -- Metriche sessione soc_start FLOAT, soc_end FLOAT, km_start FLOAT, km_driven FLOAT, duration_s INT, kwh_used FLOAT, kwh_per_100km FLOAT, -- Fingerprint CAN (per auto-detect) can_fingerprint JSONB, -- AI ai_diagnosis JSONB, -- report completo Claude fine sessione -- Timestamps started_at TIMESTAMPTZ NOT NULL, ended_at TIMESTAMPTZ, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Vista sessioni con conteggio DTC (usata dalla lista) CREATE VIEW sessions_summary AS SELECT s.*, COUNT(d.id) FILTER (WHERE d.is_active) AS dtc_active_count, COUNT(d.id) FILTER (WHERE d.severity='critical') AS dtc_critical_count, COUNT(b.id) AS battery_readings_count FROM sessions s LEFT JOIN dtc_readings d ON d.session_id = s.id LEFT JOIN battery_readings b ON b.session_id = s.id GROUP BY s.id ORDER BY s.started_at DESC; -- Indice per paginazione veloce CREATE INDEX idx_sessions_started ON sessions(started_at DESC) WHERE status = 'completed';
// web/hooks/useSessions.ts export function useSessions() { const [sessions, setSessions] = useState<Session[]>([]) const [activeSession, setActive] = useState<Session | null>(null) const supabase = createClientComponentClient() useEffect(() => { // 1. Carica ultime 50 sessioni const load = async () => { const { data } = await supabase .from('sessions_summary') // usa la VIEW .select('*') .order('started_at', { ascending: false }) .limit(50) setSessions(data ?? []) setActive(data?.find(s => s.status === 'active') ?? null) } load() // 2. Realtime: nuova sessione aperta o chiusa const channel = supabase .channel('sessions-changes') .on('postgres_changes', { event: '*', // INSERT e UPDATE schema: 'public', table: 'sessions' }, payload => { if (payload.eventType === 'INSERT') { setSessions(prev => [payload.new as Session, ...prev]) setActive(payload.new as Session) } if (payload.eventType === 'UPDATE') { setSessions(prev => prev.map(s => s.id === payload.new.id ? { ...s, ...payload.new } : s )) if (payload.new.status === 'completed') setActive(null) } }) .subscribe() return () => { supabase.removeChannel(channel) } }, []) return { sessions, activeSession } }

Accensione → ESP32 rileva ignition CAN msg 0x354
ESP32 monitora CAN bus a riposo. Quando arriva messaggio di ignition on, genera UUID sessione localmente, legge SOC e odometro iniziali via UDS.
2
POST /api/sessions/open Vercel Edge
Crea riga in Supabase con status=active. Supabase Realtime notifica la dashboard → banner "Sessione live" appare automaticamente.
3
Loop lettura ogni 1s FreeRTOS task
Task separato legge batteria + DTC e fa POST a /api/ingest. Dati salvati in battery_readings e dtc_readings con session_id.
4
Spegnimento → POST /api/sessions/close calcolo stats
Server calcola km_driven, kwh_used, kwh_per_100km. Aggiorna sessione a status=completed. Trigger automatico analisi AI Claude.
5
Report AI → Supabase → Dashboard ~2s
Claude genera report sessione in JSON. Salvato in sessions.ai_diagnosis. Dashboard riceve update via Realtime e mostra il report.
