// web/app/api/ai/chat/route.ts — streaming SSE a Next.js export async function POST(req: NextRequest) { const { question, vehicle_context, history } = await req.json() const anthropicStream = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'x-api-key': process.env.ANTHROPIC_API_KEY!, 'anthropic-version': '2023-06-01', 'content-type': 'application/json' }, body: JSON.stringify({ model: 'claude-sonnet-4-20250514', max_tokens: 600, stream: true, // ← abilita streaming SSE ...PROMPT_CHAT(vehicle_context, history, question) }) }) // Forwarda lo stream SSE direttamente al browser return new Response(anthropicStream.body, { headers: { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache', 'Connection': 'keep-alive' } }) } // web/components/AIChatBox.tsx — consuma lo stream async function sendMessage(question: string) { setTyping(true) const res = await fetch('/api/ai/chat', { method: 'POST', body: JSON.stringify({ question, vehicle_context, history }) }) const reader = res.body!.getReader() const decoder = new TextDecoder() let fullText = '' while (true) { const { done, value } = await reader.read() if (done) break const chunk = decoder.decode(value) const lines = chunk.split('\n').filter(l => l.startsWith('data:')) for (const line of lines) { const evt = JSON.parse(line.slice(5)) if (evt.type === 'content_block_delta') fullText += evt.delta.text } setStreamingText(fullText) // aggiorna UI mentre arriva } setTyping(false) saveToHistory(question, fullText) }
// web/lib/vehicle-context.ts // Costruisce il contesto veicolo per Claude da Supabase // Chiamato prima di ogni richiesta AI export async function buildVehicleContext(sessionId: string) { const supabase = createClient(url, key) // Prende ultima lettura batteria const { data: battery } = await supabase .from('battery_readings') .select('soc,soh,pack_voltage,temp_avg,temp_max,volt_delta') .eq('session_id', sessionId) .order('recorded_at', { ascending: false }) .limit(1).single() // DTC attivi const { data: dtcs } = await supabase .from('active_dtcs') // ← la VIEW creata prima .select('code,description_it,severity,system') .eq('session_id', sessionId) // Trend SOH ultime 30 sessioni const { data: sohTrend } = await supabase .from('battery_readings') .select('soh,recorded_at') .order('recorded_at', { ascending: false }) .limit(30) const sohDelta = sohTrend ? sohTrend[0].soh - sohTrend[sohTrend.length-1].soh : 0 return { ...battery, active_dtcs: dtcs?.map(d => d.code) ?? [], dtc_details: dtcs ?? [], soh_trend_30sessions: sohDelta.toFixed(2), // Claude riceve tutto questo come contesto } } // Risultato usato in ogni prompt: // { soc:78, soh:91, pack_voltage:356, temp_avg:24, temp_max:27, // volt_delta:0.04, active_dtcs:['P0AA6','U0100'], // soh_trend_30sessions:'-0.42' }
// web/app/api/ingest/route.ts — logica trigger automatici // Viene chiamato ogni volta che ESP32 invia dati async function checkTriggers(data: IngestPayload) { const triggers: Promise<void>[] = [] // Trigger 1: DTC critico nuovo const newCritical = data.dtcs?.filter(d => d.severity === 'critical' && !d.was_known ) if (newCritical?.length) { triggers.push(analyzeWithClaude({ type: 'dtc_critical', session_id: data.session_id, priority: 'immediate', payload: { dtcs: newCritical } })) } // Trigger 2: SOH sotto soglia (configurabile da utente) const sohThreshold = await getConfig('soh_alert_threshold') ?? 85 if (data.battery?.soh && data.battery.soh < sohThreshold) { triggers.push(analyzeWithClaude({ type: 'soh_threshold', session_id: data.session_id, priority: 'normal', payload: { soh: data.battery.soh, threshold: sohThreshold } })) } // Trigger 3: Anomalia termica celle (delta > config) const tempDelta = data.battery?.temp_max - data.battery?.temp_avg if (tempDelta > 3) { triggers.push(analyzeWithClaude({ type: 'thermal_anomaly', session_id: data.session_id, priority: 'normal', payload: { temp_delta: tempDelta, cell_temps: data.battery.cell_temps } })) } await Promise.all(triggers) // tutti in parallelo }
-- supabase/migrations/008_ai_analyses.sql CREATE TABLE ai_analyses ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), session_id UUID REFERENCES sessions(id), trigger_type TEXT, -- 'dtc_critical','soh_threshold','thermal','manual_chat','session_end' question TEXT, -- domanda utente (se chat) answer TEXT, -- risposta Claude completa answer_json JSONB, -- risposta strutturata (per analisi DTC) urgency TEXT, -- 'immediata','settimana','monitorare','nessuna' tokens_used INT, cost_eur FLOAT, -- tokens * prezzo per monitoraggio costi model TEXT DEFAULT 'claude-sonnet-4-20250514', vehicle_snapshot JSONB, -- contesto veicolo al momento dell'analisi created_at TIMESTAMPTZ DEFAULT NOW() ); -- Indice per storico per sessione CREATE INDEX idx_ai_session ON ai_analyses(session_id, created_at DESC); -- Vista costi mensili per monitoring CREATE VIEW monthly_ai_costs AS SELECT date_trunc('month', created_at) AS month, COUNT(*) AS analyses, SUM(tokens_used) AS total_tokens, ROUND(SUM(cost_eur)::numeric, 4) AS total_eur FROM ai_analyses GROUP BY 1 ORDER BY 1 DESC;
