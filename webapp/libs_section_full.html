// scripts/converters/jeju-csv.ts — converte formato JejuSoul // Input: "003_Kona&Niro_EV_BMS.csv" (header: PID,Length,Name,Equation,Min,Max,Unit) // Output: formato interno JSON signals[] import { parse } from 'csv-parse/sync' export function convertJejuCSV(csvContent: string, vehicleId: string) { const rows = parse(csvContent, { columns: true, skip_empty_lines: true }) return rows.map((row: any) => ({ vehicle_id: vehicleId, did: row.PID, // es. "0x01F5" name: row.Name, // es. "BMS SOC" formula: row.Equation, // es. "A*0.5" unit: row.Unit, // es. "%" min_value: parseFloat(row.Min) || null, max_value: parseFloat(row.Max) || null, category: inferCategory(row.Name), source_file: 'JejuSoul/OBD-PIDs-for-HKMC-EVs' })) } // scripts/converters/dbc.ts — converte file DBC (Tesla, PSA) export function convertDBC(dbcContent: string, vehicleId: string) { // Parsing manuale DBC: cerca "BO_" (messaggi) e "SG_" (segnali) const signals = [] const lines = dbcContent.split('\n') let currentMsgId = null for (const line of lines) { if (line.startsWith('BO_')) { // BO_ 1234 BMS_Status: 8 Vector__XXX currentMsgId = '0x' + parseInt(line.split(' ')[1]).toString(16).toUpperCase() } if (line.trim().startsWith('SG_') && currentMsgId) { // SG_ SOC : 0|8@1+ (0.5,0) [0|100] "%" Vector__XXX const m = line.match( /SG_\s+(\w+)\s*:\s*(\d+)\|(\d+).*\(([^,]+),([^)]+)\).*\[([^\]]+)\]\s+"([^"]*)"/) if (m) signals.push({ vehicle_id: vehicleId, did: currentMsgId, name: m[1], formula: `A*${m[4]}+${m[5]}`, unit: m[7], category: inferCategory(m[1]), source_file: 'DBC' }) } } return signals } function inferCategory(name: string): string { const n = name.toLowerCase() if (/soc|soh|cell|batt|pack|volt|temp/.test(n)) return 'battery' if (/motor|torque|rpm|speed/.test(n)) return 'motor' if (/charg|obc|pilot|inlet/.test(n)) return 'charging' if (/can|bus|ecu|comm/.test(n)) return 'network' return 'general' }

Usa cache SPIFFS se fresca (< 7 giorni) if (SPIFFS.exists(path)) { File f = SPIFFS.open(path, "r"); uint32_t age = (millis() - f.getLastWrite()) / 1000 / 86400; f.close(); if (age < 7) { Serial.println("[LIB] Cache valida, skip download"); return parseFromSPIFFS(path); } } // 2. Scarica JSON da Vercel API HTTPClient http; String url = nvs.get("server_url") + "/api/libs/" + vehicle_id; http.begin(url); http.addHeader("X-API-Key", nvs.get("api_key")); int code = http.GET(); if (code != 200) { Serial.printf("[LIB] Download fallito: %d\n", code); return SPIFFS.exists(path) ? parseFromSPIFFS(path) : false; } // 3. Salva in SPIFFS String json = http.getString(); File f = SPIFFS.open(path, "w"); f.print(json); f.close(); http.end(); return parseFromSPIFFS(path); } bool LibLoader::parseFromSPIFFS(const String& path) { File f = SPIFFS.open(path, "r"); DynamicJsonDocument doc(32768); // 32KB per libreria deserializeJson(doc, f); f.close(); // Popola array segnali in RAM per accesso veloce durante lettura JsonArray signals = doc["signals"]; for (auto sig : signals) { Signal s; strlcpy(s.did, sig["did"], sizeof(s.did)); strlcpy(s.name, sig["name"], sizeof(s.name)); strlcpy(s.formula, sig["formula"], sizeof(s.formula)); strlcpy(s.unit, sig["unit"], sizeof(s.unit)); activeSignals.push_back(s); } Serial.printf("[LIB] %d segnali caricati\n", activeSignals.size()); return true; }
// web/app/api/libs/[vehicle_id]/route.ts // Serve libreria JSON all'ESP32 — risposta compressa gzip export async function GET( req: NextRequest, { params }: { params: { vehicle_id: string } } ) { // Auth: valida API key nell'header const apiKey = req.headers.get('x-api-key') if (apiKey !== process.env.DEVICE_API_KEY) return NextResponse.json({ error: 'unauthorized' }, { status: 401 }) const { vehicle_id } = params // Recupera segnali da Supabase const { data: vehicle } = await supabase .from('vehicles') .select('*') .eq('slug', vehicle_id) .single() if (!vehicle) return NextResponse.json({ error: 'not found' }, { status: 404 }) const { data: signals } = await supabase .from('signals') .select('did,name,formula,unit,category,ecu_address,min_value,max_value') .eq('vehicle_id', vehicle.id) const { data: ecus } = await supabase .from('ecus').select('name,address') .eq('vehicle_id', vehicle.id) const lib = { vehicle_id, make: vehicle.make, model: vehicle.model, can_speed: vehicle.can_speed, protocol: vehicle.protocol, ecus, signals, generated_at: new Date().toISOString() } // Cache Edge per 24h — evita query Supabase ad ogni boot ESP32 return NextResponse.json(lib, { headers: { 'Cache-Control': 'public, max-age=86400, stale-while-revalidate' } }) } // web/app/api/libs/detect/route.ts — riconosce veicolo da fingerprint CAN export async function POST(req: NextRequest) { const { can_ids } = await req.json() // can_ids: ["0x354","0x7EC","0x18DAF1EF",...] const { data: vehicles } = await supabase .from('vehicles') .select('id,slug,make,model,can_fingerprint') // Trova il veicolo con più CAN ID in comune const best = vehicles ?.map(v => ({ ...v, score: v.can_fingerprint?.filter((id: string) => can_ids.includes(id)).length ?? 0 })) .sort((a, b) => b.score - a.score)[0] if (!best || best.score < 3) return NextResponse.json({ detected: false }) return NextResponse.json({ detected: true, vehicle: best }) }
